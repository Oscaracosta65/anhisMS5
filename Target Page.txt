function leFmtDate(?string $date): string
{
    if (!$date) return '';
    $ts = strtotime($date);
    return ($ts === false) ? '' : date('m-d-Y', $ts);
}

function lePad2($v): string
{
    $v = trim((string) $v);
    if ($v === '') return '';
    if (ctype_digit($v) && (int) $v < 10) return str_pad($v, 2, '0', STR_PAD_LEFT);
    return $v;
}

function leResolveLogo(string $stateAbrev, string $gName, string $gameId): array
{
    $overrides = [
        // 'MS5' => '/images/lottodb/us/ms/mississippi-match-5.png',
    ];

    $stateSlug   = strtolower(trim($stateAbrev));
    $lotterySlug = strtolower(trim($gName));
    $lotterySlug = str_replace(' ', '-', $lotterySlug);

    $rel = '';
    if ($gameId !== '' && isset($overrides[$gameId])) {
        $rel = (string) $overrides[$gameId];
    } else {
        $rel = '/images/lottodb/us/' . $stateSlug . '/' . $lotterySlug . '.png';
    }

    $abs = rtrim(JPATH_ROOT, DIRECTORY_SEPARATOR) . str_replace('/', DIRECTORY_SEPARATOR, $rel);
    if ($rel !== '' && is_file($abs)) {
        return ['url' => $rel, 'exists' => true];
    }

    return ['url' => '', 'exists' => false];
}

function leFetchDraws(\Joomla\Database\DatabaseDriver $db, string $dbCol, string $gameId, int $limit): array
{
    if ($dbCol === '' || $gameId === '' || $limit <= 0) return [];

    $q = $db->getQuery(true)
        ->select([
            $db->quoteName('draw_date'),
            $db->quoteName('next_draw_date'),
            $db->quoteName('next_jackpot'),
            $db->quoteName('first'),
            $db->quoteName('second'),
            $db->quoteName('third'),
            $db->quoteName('fourth'),
            $db->quoteName('fifth'),
            $db->quoteName('sixth'),
        ])
        ->from($db->quoteName($dbCol))
        ->where($db->quoteName('game_id') . ' = ' . $db->quote($gameId))
        ->order($db->quoteName('draw_date') . ' DESC');

    $db->setQuery($q, 0, $limit);
    $rows = $db->loadAssocList();
    return is_array($rows) ? $rows : [];
}

function leInitRange(int $min, int $max): array
{
    $counts = [];
    $seenIdx = [];
    for ($i = $min; $i <= $max; $i++) {
        $k = ($i < 10) ? '0' . $i : (string) $i;
        $counts[$k] = 0;
        $seenIdx[$k] = null;
    }
    return [$counts, $seenIdx];
}

function leDrawingsAgoLabel(?int $idx, int $window): array
{
    if ($idx === null) return [0, "Not in last {$window} drws"];
    if ($idx === 0) return [1, "In last drw"];
    return [$idx + 1, ($idx + 1) . " drws ago"];
}

function leComputeCounts(array $rows, int $min, int $max): array
{
    [$counts, $seen] = leInitRange($min, $max);

    foreach ($rows as $idx => $r) {
        $balls = [
            trim((string)($r['first'] ?? '')),
            trim((string)($r['second'] ?? '')),
            trim((string)($r['third'] ?? '')),
            trim((string)($r['fourth'] ?? '')),
            trim((string)($r['fifth'] ?? '')),
        ];
        foreach ($balls as $b) {
            if ($b === '') continue;
            if (!isset($counts[$b])) continue;
            $counts[$b]++;
            if ($seen[$b] === null) $seen[$b] = (int) $idx;
        }
    }

    return [$counts, $seen];
}

<script type="text/javascript">
(function(){
  var labelsAll = <?php echo json_encode($chartLabels, JSON_HEX_TAG|JSON_HEX_AMP|JSON_HEX_APOS|JSON_HEX_QUOT); ?>;
  var data100   = <?php echo json_encode($chartVals100, JSON_HEX_TAG|JSON_HEX_AMP|JSON_HEX_APOS|JSON_HEX_QUOT); ?>;
  var dataBack  = <?php echo json_encode($chartValsBack, JSON_HEX_TAG|JSON_HEX_AMP|JSON_HEX_APOS|JSON_HEX_QUOT); ?>;

  var back = <?php echo (int)$back; ?>;
  var nod  = <?php echo (int)$nod; ?>;

  function loadChartJsIfNeeded(done){
    if (window.Chart) { done(); return; }
    var s = document.createElement('script');
    s.src = 'https://cdn.jsdelivr.net/npm/chart.js@4.4.1/dist/chart.umd.min.js';
    s.async = true;
    s.onload = function(){ done(); };
    s.onerror = function(){ done(); };
    document.head.appendChild(s);
  }

  function setPressed(btnA, btnB, aPressed){
    if (!btnA || !btnB) return;
    btnA.setAttribute('aria-pressed', aPressed ? 'true' : 'false');
    btnB.setAttribute('aria-pressed', aPressed ? 'false' : 'true');
  }

  function render(){
    if (!window.Chart) return;

    var canvas = document.getElementById('freqChartAll');
    if (!canvas) return;

    var subtitleEl = document.getElementById('freqChartSubtitle');
    var btn100  = document.getElementById('btnChart100');
    var btnBack = document.getElementById('btnChartBack');

    var activeKey = 'BACK';
    var chart;

    try {
      chart = new Chart(canvas.getContext('2d'), {
        type: 'bar',
        data: { labels: labelsAll, datasets: [{ label: 'Drawn Times', data: dataBack, borderWidth: 0 }] },
        options: {
          responsive: true,
          maintainAspectRatio: false,
          plugins: { legend: { display: false }, tooltip: { enabled: true } },
          scales: {
            x: { ticks: { autoSkip: true, maxTicksLimit: 14 }, grid: { display: false } },
            y: { beginAtZero: true, ticks: { precision: 0 }, grid: { color: 'rgba(10,26,51,.08)' } }
          }
        }
      });
    } catch (e) { return; }

    function applyDataset(key){
      activeKey = key;
      chart.data.datasets[0].data = (key === '100') ? data100 : dataBack;
      chart.update();

      if (subtitleEl) {
        subtitleEl.textContent = (key === '100')
          ? 'Showing: Past 100 drawings'
          : ('Showing: Earlier window • Back ' + String(back) + ' • N=' + String(nod));
      }
      setPressed(btn100, btnBack, key === '100');
    }

    if (btn100)  btn100.addEventListener('click', function(){ if (activeKey !== '100') applyDataset('100'); });
    if (btnBack) btnBack.addEventListener('click', function(){ if (activeKey !== 'BACK') applyDataset('BACK'); });

    applyDataset('BACK');
  }

  loadChartJsIfNeeded(function(){
    if (document.readyState === 'loading') document.addEventListener('DOMContentLoaded', render);
    else render();
  });
})();
</script>
